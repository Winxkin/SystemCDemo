82 tlm::tlm_sync_enum nb_transport_fw(tlm::tlm_generic_payload& trans, tlm::tlm_phase& phase, sc_core::sc_time& delay) {
83 switch (phase)
84 {
85 case tlm::BEGIN_REQ:
86 {
87 // Handle BEGIN_REQ phase
88 if (m_message)
89 {
90 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " BEGIN_REQ received" << std::endl;
91 }
92 switch (trans.get_command()) {
93 case tlm::TLM_WRITE_COMMAND:
94 {
95 unsigned int wdata = 0;
96 std::memcpy(&wdata, trans.get_data_ptr(), sizeof(wdata));
97 regs.update_register(trans.get_address(), wdata);
98 if (m_message)
99 {
100 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Received transaction with address 0x" << std::hex << trans.get_address() << " data: 0x" << std::hex << wdata << std::dec << std::endl;
101 }
102 trans.set_response_status(tlm::TLM_OK_RESPONSE);
103 break;
104 }
105 case tlm::TLM_READ_COMMAND:
106 {
107 unsigned int value = regs[trans.get_address()].get_value();
108 unsigned char* cvalue = reinterpret_cast<unsigned char*>(&value);
109 memcpy(trans.get_data_ptr(), cvalue, trans.get_data_length());
110 trans.set_response_status(tlm::TLM_OK_RESPONSE);
111 break;
112 }
113 default:
114 break;
115 }
116
117 /*
118 Return data and END_REQ to bus
119 */
120 tlm::tlm_phase next_phase = tlm::END_REQ;
121 target_socket->nb_transport_bw(trans, next_phase, delay);
122 break;
123 }
124 case tlm::END_REQ:
125 {
126 // Handle END_REQ phase (shouldn't happen here)
127 if (m_message)
128 {
129 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " END_REQ received" << std::endl;
130 }
131 break;
132 }
133 default:
134 break;
135 }
136 return tlm::TLM_ACCEPTED;
137 }

180 void thr_triggered_port_process()
181 {
