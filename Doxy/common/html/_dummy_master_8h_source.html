38 /*
39 * nb_transport_bw
40 *
41 * Implements the non-blocking backward transport interface for the initiator.
42 *
43 * @param trans Reference to the generic payload object containing the transaction details
44 * such as command, address, and data.
45 *
46 * @param phase Reference to the transaction phase. The current phase of the transaction,
47 * which may be updated by the function.
48 *
49 * @param delay Reference to the annotated delay. Specifies the timing delay for the transaction
50 * and may be updated by the function.
51 *
52 * @return tlm::tlm_sync_enum Enumeration indicating the synchronization state of the transaction:
53 * - TLM_ACCEPTED: Transaction is accepted, and no immediate further action is required.
54 * - TLM_UPDATED: Transaction phase has been updated. The initiator should check the new phase.
55 * - TLM_COMPLETED: Transaction is completed immediately, and no further phases will occur.
56 */
57 tlm::tlm_sync_enum nb_transport_bw(tlm::tlm_generic_payload& trans, tlm::tlm_phase& phase, sc_core::sc_time& delay) {
58 switch (phase)
59 {
60 case tlm::BEGIN_REQ:
61 {
62 // Handle BEGIN_REQ phase
63 if (m_message)
64 {
65 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " BEGIN_REQ received" << std::endl;
66 }
67 break;
68 }
69 case tlm::END_REQ:
70 {
71 // Handle END_REQ phase (shouldn't happen here)
72 if (m_message)
73 {
74 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " END_REQ received" << std::endl;
75 }
76 if (trans.is_response_error()) {
77 if (m_message)
78 {
79 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Transaction failed with response status: " << trans.get_response_string() << std::endl;
80 }
81 }
82 else {
83 switch (trans.get_command()) {
84 case tlm::TLM_READ_COMMAND:
85 {
86 tempdata.m_data = new unsigned char[trans.get_data_length()];
87 std::memcpy(tempdata.m_data, trans.get_data_ptr(), trans.get_data_length());
88
89 // dump message
90 if (m_message)
91 {
92 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Received data: ";
93
94 for (unsigned int i = 0; i < trans.get_data_length(); i++)
95 {
96 std::cout << "[0x" << std::hex << (unsigned int)tempdata.m_data[i] << "] ";
97 }
98 }
99 std::cout << std::endl;
100
101 trans.set_response_status(tlm::TLM_OK_RESPONSE);
102 break;
103 }
104 default:
105 break;
106 }
107 if (m_message)
108 {
109 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Transaction succeeded" << std::endl;
110 }
111 }
112 break;
113 }
114 default:
115 break;
116 };
117 return tlm::TLM_ACCEPTED;
118 }

185 /*
186 * SentTransaction
187 *
188 * Implements the sent a transaction with 32 bit data to bus MMIO
189 *
190 * @param addr Reference to the address of the slave
191 *
192 * @param data Reference to 32-bit data
193 *
194 * @param cmd Reference to tlm command
195 * -TLM_READ_COMMAND : TLM read requset
196 * -TLM_WRITE_COMMAND : TLM write requset
197 * -TLM_IGNORE_COMMAND: TLM ignore requset
198 */
199 void SentTransaction(unsigned int addr, uint32_t data, tlm::tlm_command cmd)
200 {
201 switch (cmd) {
202 case tlm::TLM_WRITE_COMMAND:
203 {
204 tlm::tlm_generic_payload trans;
205 unsigned char* _data = reinterpret_cast<unsigned char*>(&data);
206 sc_core::sc_time delay = sc_core::SC_ZERO_TIME;
207 trans.set_command(tlm::TLM_WRITE_COMMAND);
208 trans.set_address(addr);
209 trans.set_data_length(sizeof(data));
210 trans.set_data_ptr(_data);
211 trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
212 if (m_message)
213 {
214 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Sending Write transaction with address 0x" << std::hex << addr

216 }
217
218 tlm::tlm_phase phase = tlm::BEGIN_REQ;
219 tlm::tlm_sync_enum status;
220 status = initiator_socket->nb_transport_fw(trans, phase, delay);
221
222 break;
223
224 }
225 case tlm::TLM_READ_COMMAND:
226 {
227 tlm::tlm_generic_payload trans;
228 unsigned int data;
229 sc_core::sc_time delay = sc_core::SC_ZERO_TIME;

231 trans.set_command(tlm::TLM_READ_COMMAND);
