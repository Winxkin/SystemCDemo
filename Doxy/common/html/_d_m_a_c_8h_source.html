83 /*
84 * nb_transport_fw
85 *
86 * Implements the non-blocking backward transport interface for the initiator.
87 *
88 * @param trans Reference to the generic payload object containing the transaction details
89 * such as command, address, and data.
90 *
91 * @param phase Reference to the transaction phase. The current phase of the transaction,
92 * which may be updated by the function.
93 *
94 * @param delay Reference to the annotated delay. Specifies the timing delay for the transaction
95 * and may be updated by the function.
96 *
97 * @return tlm::tlm_sync_enum Enumeration indicating the synchronization state of the transaction:
98 * - TLM_ACCEPTED: Transaction is accepted, and no immediate further action is required.
99 * - TLM_UPDATED: Transaction phase has been updated. The initiator should check the new phase.
100 * - TLM_COMPLETED: Transaction is completed immediately, and no further phases will occur.
101 */
102 tlm::tlm_sync_enum nb_transport_fw(tlm::tlm_generic_payload& trans, tlm::tlm_phase& phase, sc_core::sc_time& delay) {
103 switch (phase)
104 {
105 case tlm::BEGIN_REQ:
106 {
107 // Handle BEGIN_REQ phase
108 if (m_message)
109 {
110 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " (Target socket) BEGIN_REQ received" << std::endl;
111 }
112 switch (trans.get_command()) {
113 case tlm::TLM_WRITE_COMMAND:
114 {
115 unsigned int wdata = 0;
116 std::memcpy(&wdata, trans.get_data_ptr(), sizeof(wdata));
117 regs.update_register(trans.get_address(), wdata);
118 if (m_message)
119 {
120 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Received transaction with address 0x" << std::hex << trans.get_address() << " data: 0x" << std::hex << wdata << std::dec << std::endl;
121 }
122 trans.set_response_status(tlm::TLM_OK_RESPONSE);
123 break;
124 }
125 case tlm::TLM_READ_COMMAND:
126 {
127 unsigned int value = regs[trans.get_address()].get_value();
128 unsigned char* cvalue = reinterpret_cast<unsigned char*>(&value);
129 memcpy(trans.get_data_ptr(), cvalue, trans.get_data_length());
130 trans.set_response_status(tlm::TLM_OK_RESPONSE);
131 break;
132 }
133 default:
134 break;
135 }
136
137 /*
138 Return data and END_REQ to bus
139 */
140 tlm::tlm_phase next_phase = tlm::END_REQ;
141 target_socket->nb_transport_bw(trans, next_phase, delay);
142 break;
143 }
144 case tlm::END_REQ:
145 {
146 // Handle END_REQ phase (shouldn't happen here)
147 if (m_message)
148 {
149 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " (Target socket) END_REQ received" << std::endl;
150 }
151 break;
152 }
153 default:
154 break;
155 }
156 return tlm::TLM_ACCEPTED;
157 }

159 /*
160 * nb_transport_bw
161 *
162 * Implements the non-blocking backward transport interface for the initiator.
163 *
164 * @param trans Reference to the generic payload object containing the transaction details
165 * such as command, address, and data.
166 *
167 * @param phase Reference to the transaction phase. The current phase of the transaction,
168 * which may be updated by the function.
169 *
170 * @param delay Reference to the annotated delay. Specifies the timing delay for the transaction
171 * and may be updated by the function.
172 *
173 * @return tlm::tlm_sync_enum Enumeration indicating the synchronization state of the transaction:
174 * - TLM_ACCEPTED: Transaction is accepted, and no immediate further action is required.
175 * - TLM_UPDATED: Transaction phase has been updated. The initiator should check the new phase.
176 * - TLM_COMPLETED: Transaction is completed immediately, and no further phases will occur.
177 */
178 tlm::tlm_sync_enum nb_transport_bw(tlm::tlm_generic_payload& trans, tlm::tlm_phase& phase, sc_core::sc_time& delay) {
179 switch (phase)
180 {
181 case tlm::BEGIN_REQ:
182 {
183 // Handle BEGIN_REQ phase
184 if (m_message)
185 {
186 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " (Initiator socket) BEGIN_REQ received" << std::endl;
187 }
188 break;
189 }
190 case tlm::END_REQ:
191 {
192 // Handle END_REQ phase
193 if (m_message)
194 {
195 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " (Initiator socket) END_REQ received" << std::endl;
196 }
197 if (trans.is_response_error())
198 {
199 if (m_message)
200 {
201 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << ": Transaction failed with response status: " << trans.get_response_string() << std::endl;
202 }
203 }
204 else
205 {
206 switch (trans.get_command()) {
207 case tlm::TLM_READ_COMMAND:
208 {
209 // Change the address of transaction to destination address
210 copy_tlm_generic_payload(current_trans, trans);
211 current_trans.set_address(regs[DMADESADDR(m_current_ch)].get_value());
212 current_trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
213 current_trans.set_command(tlm::TLM_WRITE_COMMAND);
214 e_DMA_forward.notify();
215
216 break;
217 }
218 case tlm::TLM_WRITE_COMMAND:
219 {
220 if (m_message)
221 {
222 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " : DMA transfers data from 0x" << std::hex
223 << regs[DMASRCADDR(m_current_ch)].get_value() << " to 0x" << std::hex << regs[DMADESADDR(m_current_ch)].get_value() << std::endl;
224 }
225 // Notifying DMA operation done !
226 e_DMA_run_done.notify();
227 break;
228 }
229 default:
230 break;
231 }
232 }
233 break;
234 }
235 default:
236 break;
237 };
238 return tlm::TLM_ACCEPTED;
239 }

241 /*
242 * init_registers
243 *
244 * To initialize registers for DMAC model
245 *
246 */
247 void init_registers()
248 {
249 for (unsigned int i = 0; i < DMA_MAX_CH; i++)
250 {
251 regs.add_register("DMADESADDR" + std::to_string(i), DMADESADDR(i), 0x00, 0xFFFFFFFF, i, READWRITE);
252 regs.add_register("DMASRCADDR" + std::to_string(i), DMASRCADDR(i), 0x00, 0xFFFFFFFF, i, READWRITE);
253 regs.add_register("DMADATALENGTH" + std::to_string(i), DMADATALENGTH(i), 0x00, 0xFFFFFFFF, i, READWRITE);
254 }
255 for (unsigned int i = 0; i < (DMA_MAX_CH/32); i++)
256 {
257 regs.add_register("DMAREQ" + std::to_string(i), DMAREQ(i), 0x00, 0xFFFFFFFF, i, READWRITE);
258 regs.add_register("DMAACK" + std::to_string(i), DMAACK(i), 0x00, 0xFFFFFFFF, i, READWRITE);
259 regs.add_register("DMAINT" + std::to_string(i), DMAINT(i), 0x00, 0xFFFFFFFF, i, READWRITE);
260 regs.add_register("DMACHEN" + std::to_string(i), DMACHEN(i), 0x00, 0xFFFFFFFF, i, READWRITE);
261
262 // Registration call back function
263 regs.set_register_callback("DMAREQ" + std::to_string(i), std::bind(&DMAC::cb_DMAREQ, this,
264 std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4, std::placeholders::_5));
265 }

334 for (unsigned int i = 0; i < DMA_MAX_CH; i++) {

343 }

364 /*
365 * thr_priority_process
366 *
367 * Impelmentation of the thread to handle priority of DMA operation
368 *
369 */
370 void thr_priority_process()
371 {
372 while (true)
373 {
374 // waiting to next cycle
375 wait(e_DMA_request);
376 wait(clk.posedge_event());
377 if (!m_testmode)
378 {
379 // using ports to trigger DMAC
380 while (!port_req_ids.empty())
381 {
382 m_current_ch = port_req_ids.front();
383 port_req_ids.pop_front(); // removing request id from list
384 e_DMA_run.notify(); // Bus delay simulation time
385 wait(e_DMA_run_done);
386
387 // set output ack and int port
388 wait(clk.posedge_event());
389 DMA_ack[m_current_ch].write(true);
390 DMA_int[m_current_ch].write(true);
391 wait(clk.posedge_event());
392 DMA_ack[m_current_ch].write(false);
393 DMA_int[m_current_ch].write(false);
394 // continuting with next channels
395 }
396
397 }
398 else
399 {
400 // using register to trigger DMAC
401 uint32_t value = regs[m_cur_reg_name].get_value();
402 for (unsigned int i = 0; i < 32; i++)
403 {
404 if ((regs[DMACHEN(m_cur_reg_ch)].get_value() >> i) & 0x01)
405 {
406 if ((value >> i) & 0x01)
407 {
408 // set current channel
409 m_running = true;
410 m_current_ch = (m_cur_reg_ch == 0) ? i : (i * m_cur_reg_ch);
411 e_DMA_run.notify(); // Bus delay simulation time
412 wait(e_DMA_run_done);
413 // continuting with next channels
414 regs[DMAACK(m_cur_reg_ch)] = (0x01 << i) | regs[DMAACK(m_cur_reg_ch)].get_value();
415 regs[DMAINT(m_cur_reg_ch)] = (0x01 << i) | regs[DMAINT(m_cur_reg_ch)].get_value();
416 }
417 }
418 }
419 }
420 if (m_message)
421 {
422 std::cout << "[" << sc_core::sc_time_stamp().to_double() << " NS ]" << m_name << " DMA process is done" << std::endl;
423 }
424 // releasing flags
425 m_running = false;
426 m_testmode = false;
427
428 }

431 /*
432 * thr_DMA_run_process
433 *
434 * Impelmentation of the thread to start DMA operation
435 *
436 */
437 void thr_DMA_run_process()
438 {
439 while (true)
440 {
441 wait(e_DMA_run);
442 // DMA operation in here
443 /*
444 Initiator socket -> request trans -> get return data trans
445 */
446 tlm::tlm_generic_payload trans;
447 sc_core::sc_time delay = sc_core::SC_ZERO_TIME;
448 unsigned char* data = new unsigned char[regs[DMADATALENGTH(m_current_ch)].get_value()];
449 // Configuration transaction
450 trans.set_command(tlm::TLM_READ_COMMAND);
451 trans.set_address(regs[DMASRCADDR(m_current_ch)].get_value());
452 trans.set_data_ptr(data);
453 trans.set_data_length(regs[DMADATALENGTH(m_current_ch)].get_value());
454 trans.set_streaming_width(regs[DMADATALENGTH(m_current_ch)].get_value()); // = data_length to indicate no streaming
455 trans.set_byte_enable_ptr(0); // 0 indicates unused
456 trans.set_dmi_allowed(false); // DMI not allowed
457 trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
458 // Sending transaction using initiator socket
459 tlm::tlm_phase fw_phase = tlm::BEGIN_REQ;
460 tlm::tlm_sync_enum status;
461 status = initiator_socket->nb_transport_fw(trans, fw_phase, delay);
462 }
